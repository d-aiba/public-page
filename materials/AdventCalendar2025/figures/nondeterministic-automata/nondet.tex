  {\@ddfrac\scriptstyle{#1}{#2}}}

\newcommand{\@ddfrac}[3]{%
  \begingroup
  \setbox0=\hbox{$\m@th#1#2$}%
  \setbox2=\hbox{$\m@th#1#3$}%
  \dimen0=\wd0
  \ifdim\wd2>\dimen0 \dimen0=\wd2\fi
  \vcenter{%
    \offinterlineskip
    \halign{%
      \hfil$\m@th#1##$\hfil\cr
      \box0\cr
      \noalign{\hrule height0.4pt \kern1pt \hrule height0.4pt \kern1pt}
      \box2\cr
    }%
  }%
  \endgroup
}
\makeatother
